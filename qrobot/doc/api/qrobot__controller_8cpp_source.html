00066             if(data[10] == 0x01)

00071                 return HEAD_LONG_TOUCH;
00072             else if(data[10] == 0x04)
00073                 return NECK_TOUCH_DOWN;
00074             else if(data[10] == 0x08)
00075                 return NECK_TOUCH_UP;
00076             else
00077                 return UNKNOWN;
00078         }
00079         if(data[11] == 0x02)
00080             return POWER_OFF;

00315         unsigned char *send_buf, int write_len, int wait_time)
00316 {

00319     for (int i = 0; i < write_len - 1; i++)
00320     {
00321         checksum ^= send_buf[i];
